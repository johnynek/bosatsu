package Demo/Compute export fib, factorial # Fibonacci using int_loop for efficient iteration def fib(n: Int) -> Int: int_loop(n, (0, 1), (i, acc) -> (a, b) = acc (i.sub(1), (b, a.add(b)))) .fst # Factorial using int_loop def factorial(n: Int) -> Int: int_loop(n, 1, (i, acc) -> (i.sub(1), acc.times(i))).snd
